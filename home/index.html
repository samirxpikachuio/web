Example API Request 
fetch('https://api.paxsenix.biz.id/v1/chat/completions', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer COMING_SOON'
  },
  body: JSON.stringify({
    prompt: "Hello, how can you help me?",
    model: "gpt-3.5-turbo"
  })
})
.then(response => response.json())
.then(data => console.log(data));
